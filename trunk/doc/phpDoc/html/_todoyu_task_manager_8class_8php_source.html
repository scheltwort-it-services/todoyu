01122     public static function getParentDateRanges($idTask, $idProject = 0, $checkSelf = false) {
01123         $idTask     = intval($idTask);
01124         $idProject  = intval($idProject);
01125         $range      = false;

01128             $rootlineTasks  = self::getRootlineTasks($idTask);
01129 
01130 
01131             if( $checkSelf !== true ) {
01132                     // Remove element itself
01133                 array_shift($rootlineTasks);

01135 
01136                 // Check all parent elements if there is a container and use its dates for the range
01137             foreach($rootlineTasks as $task) {

01141                         'end'   => $task['date_end']
01142                     );
01143                     break;
01144                 }
01145             }
01146         }
01147 
01148             // If no container found, use project
01149         if( $range === false ) {
01150             if( $idProject !== 0 ) {
01151                 $project    = TodoyuProjectManager::getProject($idProject);
01152             } elseif( $idTask !== 0 ) {
01153                 $project    = TodoyuTaskManager::getProject($idTask);
01154             }
01155 
01156             if( isset($project) ) {
01157                 $range  = array(
01158                     'start' => $project->getStartDate(),
01159                     'end'   => $project->getEndDate()
01160                 );
